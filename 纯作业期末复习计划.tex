\[
\mathcal{B}\!\left(\tfrac{1}{n}\right)=\bigl\{\{\tfrac{1}{n}\}\bigr\}
\]

\[
\mathcal{B}(x)
=
\bigl\{\{\tfrac{1}{n}:n\ge m\}\cup\{x\}:m\in\mathbb{N}\bigr\}
